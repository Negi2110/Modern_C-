```cpp
// ============================================================
// PIMPL IDIOM (Pointer to Implementation) - INTERVIEW NOTES
// ============================================================

#include <string>
#include <iostream>
#include <memory>

// ============================================================
// SECTION 1: WHAT IS PIMPL AND WHY?
// ============================================================

// PIMPL = Pointer to IMPLementation
// Also called: "Cheshire Cat", "Compiler Firewall", "d-pointer" (Qt)
//
// THE PROBLEM without PIMPL:
//
// Person.hpp (public header - users see this):
// class Person {
// public:
//     std::string GetAttributes();
// private:
//     std::string m_name;      // users can SEE these
//     std::string m_strength;  // even though they can't ACCESS them
//     std::string m_speed;     // changing them = RECOMPILE everyone!
// };
//
// PROBLEMS:
// 1. COMPILATION DEPENDENCY:
//    -> if you add/remove/change a private member
//    -> .hpp file changes
//    -> EVERY file that includes Person.hpp must RECOMPILE
//    -> in large projects = recompiling 1000s of files for one change!
//
// 2. IMPLEMENTATION EXPOSURE:
//    -> private members are VISIBLE in the header
//    -> users can SEE your implementation details
//    -> reveals which libraries you use (e.g., including <string>)
//    -> not truly hidden, just inaccessible
//
// PIMPL SOLUTION:
// -> move ALL private members into a separate struct (pImplPerson)
// -> put ONLY a pointer to that struct in the header
// -> header NEVER changes when implementation changes
// -> true information hiding

// ============================================================
// SECTION 2: YOUR QUESTION - WHY GETTERS/SETTERS DON'T NEED Person::
// ============================================================

// YOUR QUESTION: "why getter or setter don't need Person::?"
//
// ANSWER: they DO use Person:: when defined OUTSIDE the class!
//
// std::string Person::GetAttributes()  <- HAS Person::
//                ^^^^^^^
//                scope resolution operator
//                "this GetAttributes belongs to Person class"
//
// WITHOUT Person:: -> free function (not a member of any class)
// std::string GetAttributes()    // completely different function!
//                                // not Person's member at all
//
// INSIDE the class body (in .hpp declaration):
// class Person {
//     std::string GetAttributes();  // NO Person:: needed
//                                   // already INSIDE Person's scope
// };
//
// OUTSIDE the class body (in .cpp implementation):
// std::string Person::GetAttributes() { }  // NEEDS Person::
//                                          // outside Person's scope
//                                          // must tell compiler which class
//
// THIS IS THE SAME RULE from your very first class session:
// Student::Student() { }     // constructor defined outside class
// Student::~Student() { }    // destructor defined outside class
// void Student::printName()  // method defined outside class
// ALL need ClassName:: when defined OUTSIDE the class body

// ============================================================
// SECTION 3: PIMPL STRUCTURE - THE THREE FILES
// ============================================================

// FILE 1: Person.hpp (PUBLIC HEADER - users see ONLY this)
// --------------------------------------------------------
// class Person {
// public:
//     Person(std::string s);
//     ~Person();
//     std::string GetAttributes();
// private:
//     struct pImplPerson;  // FORWARD DECLARATION only
//                          // users see the NAME but NOT the contents
//                          // compiler only needs to know it EXISTS
//                          // not what's INSIDE it
//     std::unique_ptr<pImplPerson> m_pImplUnique;
//                                    // pointer = 8 bytes
//                                    // regardless of what pImplPerson contains
// };
//
// KEY INSIGHT:
// -> pImplPerson is forward-declared (incomplete type in header)
// -> users have NO IDEA what m_name, m_strength, m_speed even exist
// -> add/remove/change private members -> Person.hpp NEVER changes
// -> no recompilation cascade!

// FILE 2: Person.cpp (IMPLEMENTATION - users NEVER see this)
// --------------------------------------------------------
// struct Person::pImplPerson {
//     std::string m_name;
//     std::string m_strength;
//     std::string m_speed;
//     // all implementation details COMPLETELY hidden here
// };
//
// Person::Person(std::string s) {
//     m_pImplUnique = std::make_unique<pImplPerson>(s, "strong", "fast");
// }
// Person::~Person() { }   // must be in .cpp where pImplPerson is complete!
// std::string Person::GetAttributes() {
//     return m_pImplUnique->m_name + ...;
// }

// ============================================================
// SECTION 4: RAW POINTER vs UNIQUE_PTR IN PIMPL
// ============================================================

// OLD WAY (raw pointer - your commented out code):
// private:
//     pImplPerson* m_pImpl;   // raw pointer
//
// Person::Person(std::string s) {
//     m_pImpl = new pImplPerson(s, "strong", "fast");  // manual new
// }
// Person::~Person() {
//     delete m_pImpl;    // MUST remember to delete manually!
// }
// PROBLEM: must manually write destructor, copy ctor, copy assign
//          if you forget delete -> memory leak
//          Rule of Three applies!
//
// MODERN WAY (unique_ptr - your actual code):
// private:
//     std::unique_ptr<pImplPerson> m_pImplUnique;
//
// Person::Person(std::string s) {
//     m_pImplUnique = std::make_unique<pImplPerson>(s, "strong", "fast");
// }
// Person::~Person() { }   // EMPTY! unique_ptr auto-deletes pImplPerson
//                          // RAII handles cleanup automatically
// BENEFIT:
// -> no manual delete -> no memory leak possible
// -> Rule of Zero applies (mostly)
// -> BUT: unique_ptr<pImplPerson> needs complete type at destructor point
//         THIS is why ~Person() must be defined in .cpp (not inline in .hpp)
//         where pImplPerson is fully defined

// ============================================================
// SECTION 5: WHY ~Person() MUST BE IN .cpp - IMPORTANT DETAIL
// ============================================================

// unique_ptr<T>'s destructor needs to call delete on T
// delete T requires T to be a COMPLETE TYPE (fully defined)
//
// In Person.hpp:
// struct pImplPerson;  <- INCOMPLETE type (forward declaration only)
// std::unique_ptr<pImplPerson> m_pImplUnique;
//
// If ~Person() is inline in .hpp (or compiler-generated there):
// -> ~unique_ptr<pImplPerson> tries to delete pImplPerson
// -> pImplPerson is INCOMPLETE at that point
// -> COMPILER ERROR: "cannot delete pointer to incomplete type"
//
// FIX: declare ~Person() in .hpp, define it in .cpp:
// Person.hpp: ~Person();           // declared (not inline)
// Person.cpp: Person::~Person() {} // defined where pImplPerson is complete
//
// THIS IS WHY your destructor body is empty in .cpp
// unique_ptr handles the actual deletion automatically
// you just needed to DEFINE it in .cpp (not inline) so the
// compiler can see pImplPerson's full definition

// ============================================================
// SECTION 6: PIMPL WITH unique_ptr - COPY SEMANTICS
// ============================================================

// unique_ptr disables copying:
// Person p1("John");
// Person p2 = p1;   // COMPILER ERROR!
//                   // unique_ptr copy constructor is deleted
//
// IF you want Person to be copyable with PIMPL:
// OPTION 1: use shared_ptr instead of unique_ptr
//   -> both p1 and p2 share same pImplPerson
//   -> modifying p1 ALSO modifies p2 (shallow copy)
//
// OPTION 2: write custom copy constructor in .cpp:
// Person::Person(const Person& rhs)
//     : m_pImplUnique(std::make_unique<pImplPerson>(*rhs.m_pImplUnique))
// { }  // deep copy: new pImplPerson with same values
//
// Qt uses PIMPL extensively:
// QWidget, QPainter, etc. all use "d-pointer" (their name for pImplPerson)
// accessed via Q_D() macro
// Q_D(QWidget) = qobject_cast<QWidgetPrivate*>(d_ptr.data())

// ============================================================
// SECTION 7: BENEFITS SUMMARY
// ============================================================

// BENEFIT 1: COMPILATION FIREWALL
// -> Change private implementation -> ONLY Person.cpp recompiles
// -> All files including Person.hpp: ZERO recompilation needed
// -> Critical in large codebases (saves minutes/hours of build time)
//
// BENEFIT 2: TRUE INFORMATION HIDING
// -> Private members completely invisible to users (in .cpp only)
// -> Header shows ONLY the public interface
// -> Implementation details truly encapsulated
//
// BENEFIT 3: BINARY COMPATIBILITY
// -> Can change implementation without changing ABI (binary interface)
// -> Ship new Person.cpp without recompiling client code
// -> Critical for shared libraries (.dll/.so)
//
// BENEFIT 4: REDUCED HEADER DEPENDENCIES
// -> Person.hpp only needs: <string>, <memory>
// -> Person.cpp can include: <database>, <network>, <anything>
// -> Those extra includes don't leak into client code

// ============================================================
// SECTION 8: PIMPL VISUAL
// ============================================================

// WITHOUT PIMPL:
// Person.hpp visible to user:
// ┌────────────────────────────┐
// │ class Person {             │
// │   string GetAttributes();  │
// │   string m_name;     <--VISIBLE but private  │
// │   string m_strength; <--VISIBLE but private  │
// │   string m_speed;    <--VISIBLE but private  │
// │ }                          │
// └────────────────────────────┘
//
// WITH PIMPL:
// Person.hpp visible to user:
// ┌────────────────────────────┐
// │ class Person {             │
// │   string GetAttributes();  │
// │   struct pImplPerson;      │  <- just a name, nothing else
// │   unique_ptr<pImplPerson>  │  <- opaque pointer, 8 bytes
// │ }                          │
// └────────────────────────────┘
//
// Person.cpp (user never sees):
// ┌────────────────────────────┐
// │ struct pImplPerson {       │
// │   string m_name;           │  <- truly hidden
// │   string m_strength;       │  <- truly hidden
// │   string m_speed;          │  <- truly hidden
// │ }                          │
// └────────────────────────────┘

// ============================================================
// SECTION 9: INTERVIEW QUESTIONS
// ============================================================

// Q1: What is the PIMPL idiom?
//     -> Pointer to Implementation. Moves private members into a
//        separate struct defined in the .cpp file. The class header
//        contains only a pointer to this hidden struct. Provides
//        compilation firewall and true information hiding.

// Q2: What problem does PIMPL solve?
//     -> Compilation dependency: changes to private members normally
//        require recompiling all files that include the header.
//        PIMPL breaks this dependency - only the .cpp recompiles.

// Q3: Why must ~Person() be defined in the .cpp (not inline)?
//     -> unique_ptr's destructor needs a complete type to call delete.
//        pImplPerson is incomplete in the header (forward declared).
//        Defining ~Person() in .cpp where pImplPerson is fully defined
//        allows unique_ptr to correctly delete it.

// Q4: Why use unique_ptr instead of raw pointer in PIMPL?
//     -> unique_ptr auto-deletes pImplPerson when Person is destroyed.
//        No manual delete needed. RAII ensures no memory leaks.
//        Rule of Zero: no need to write custom destructor body.

// Q5: Why do member functions defined outside class need ClassName::?
//     -> They are defined outside the class scope. The compiler needs
//        to know which class they belong to. :: (scope resolution)
//        associates the function with the correct class.

// Q6: Does PIMPL affect runtime performance?
//     -> Slight overhead: one extra heap allocation (for pImplPerson)
//        and one extra pointer dereference per member access.
//        Usually negligible but exists. Trade-off: compile-time vs runtime.

// Q7: How does Qt use PIMPL?
//     -> Qt calls it "d-pointer" pattern. Every QObject-derived class
//        has a private d_ptr pointer. Accessed via Q_D() macro.
//        Allows Qt to change implementation without breaking ABI.

// Q8: Can a class with PIMPL and unique_ptr be copied?
//     -> Not by default (unique_ptr deletes copy constructor).
//        Must write custom copy constructor in .cpp that
//        creates a new pImplPerson with copied values (deep copy).
//        Or use shared_ptr for shared ownership (shallow copy).

int main()
{
    Person p("John");
    std::cout << p.GetAttributes() << std::endl;
    // prints: "John is strong and fast"
    return 0;
}
```

**PIMPL in one picture:**

```
WITHOUT PIMPL:                    WITH PIMPL:
Person.hpp (users see):           Person.hpp (users see):
  private:                          private:
    string m_name;     ← visible!     struct pImplPerson; ← opaque!
    string m_strength; ← visible!     unique_ptr<pImplPerson>; ← 8 bytes
    string m_speed;    ← visible!
                                    Person.cpp (users NEVER see):
Change m_name type →                  struct pImplPerson {
RECOMPILE EVERYONE                      string m_name;     ← truly hidden
                                        string m_strength; ← truly hidden
                                      };
                                    Change anything → ONLY .cpp recompiles
```

> **Qt connection**: in ISSST you're using Qt classes like `QWidget`, `QTimer`, `QUdpSocket` — every single one of them uses PIMPL internally (called d-pointer). This is exactly how Qt maintains binary compatibility across versions — your compiled code works with a new Qt DLL even though Qt's internals changed, because the header (and therefore the ABI) never changed.
